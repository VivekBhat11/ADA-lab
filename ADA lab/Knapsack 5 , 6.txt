#include <stdio.h>

int max(int a, int b) {
    return (a > b) ? a : b;
}

int main() {
    int p[20], w[20], v[20][20];
    int i, j, n, W;

    printf("Enter the number of items: ");
    scanf("%d", &n);

    for (i = 1; i <= n; i++) {
        printf("Enter the weight and profit of item %d: ", i);
        scanf("%d %d", &w[i], &p[i]);
    }

    printf("\nEnter the capacity of the knapsack: ");
    scanf("%d", &W);

    for (i = 0; i <= n; i++) {
        v[i][0] = 0;
    }
    for (j = 0; j <= W; j++) {
        v[0][j] = 0;
    }

    for (i = 1; i <= n; i++) {
        for (j = 1; j <= W; j++) {
            if (w[i] > j) {
                v[i][j] = v[i - 1][j];
            } else {
                v[i][j] = max(v[i - 1][j], v[i - 1][j - w[i]] + p[i]);
            }
        }
    }

    printf("The maximum profit is %d\n", v[n][W]);

    printf("\nThe items selected are:\n");
    j = W;
    for (i = n; i >= 1; i--) {
        if (v[i][j] != v[i - 1][j]) {
            printf("\tItem: %d\n", i);
            j = j - w[i];
        }
    }

    return 0;
}
